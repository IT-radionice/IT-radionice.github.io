pobeda

  • Ukoliko je igrač povezao dva polja, pobedio je

def je_pobedio(tabla, znak):
	return ((tabla[6] == znak and tabla[7] == znak and tabla[8] == znak) or # gornja horizontala
		(tabla[3] == znak and tabla[4] == znak and tabla[5] == znak) or # srednja horizontala
		(tabla[0] == znak and tabla[1] == znak and tabla[2] == znak) or # donja horizontala
		(tabla[6] == znak and tabla[3] == znak and tabla[0] == znak) or # znakva vertikala
		(tabla[7] == znak and tabla[4] == znak and tabla[1] == znak) or # srednja vertikala
		(tabla[8] == znak and tabla[5] == znak and tabla[2] == znak) or # desna vertikala
		(tabla[6] == znak and tabla[4] == znak and tabla[2] == znak) or # dijagonala
		(tabla[8] == znak and tabla[4] == znak and tabla[0] == znak)) # dijagonala

Potez

  • Preuzima se pozicija sa tastatre
  • Karakter se upiše u tablu na tu poziciju
  • Proveri se da li je igra završena (pobeda ili popunjena tabla)

def potez(tabla, znak):
	print 'igra '+znak
	pozicija = preuzmi_poziciju() #preuzimanje pozicije
	#sve dok se ne unese slobodno mesto
	while not je_slobodno_mesto(tabla,pozicija): 
		pozicija = preuzmi_poziciju()
	upisi_poziciju(tabla, pozicija, znak) #upis karaktera na poziciju
	prikazi_tablu(tabla)
	if je_pobedio(tabla, znak): #pobeda
		print 'pobedio je '+znak
		return True
	if je_kraj(tabla): #popunjena tabla
		print 'nema pobednika'
		return True
	return False

  • Za svako slobodno mesto na tabli
  • Napravimo kopiju table
  • U kopiju upišemo znak na to slobodno mesto
  • Proverimo da li smo pobedili
  • Ako jesmo, vratimo poziciju tog mesta
  • Zašto nam treba kopija table?

	# 1. Pobeda
	for i in range(8):
		if je_slobodno_mesto(tabla,i):
			kopija = kopiraj_tablu(tabla)
			upisi_poziciju(kopija,i, znak)
			if je_pobedio(kopija, znak):
				return i

2. Blokiranje

  • Za svako slobodno mesto na tabli
  • Napravimo kopiju table
  • U kopiju upišemo protivnikov znak na to slobodno mesto
  • Proverimo da li je protivnik pobedio
  • Ako jeste, vratimo poziciju tog mesta

	# 2. Blokiranje
	for i in range(8):
		if je_slobodno_mesto(tabla,i):
			kopija = kopiraj_tablu(tabla)
			upisi_poziciju(kopija,i, protivnicki_znak)
			if je_pobedio(kopija, protivnicki_znak):
				return i

3. Uglovi

  • Proverimo da li postoji slobodan ugao
  • Ako postoji, vratimo njegovu poziciju

	# 3. Uglovi
	for i in [0,2,6,8]:
		if je_slobodno_mesto(tabla,i):
			return i

4. Centar

  • Proverimo da li je slobodan centar
  • Ako jeste, vratimo njegovu poziciju

	# 4. Centar
	if je_slobodno_mesto(tabla,4):
		return 4

5. Stranice

  • Proverimo da li je slobodno mesto na nekoj od stranica
  • Ako jeste, vratimo njegovu poziciju

	# 5. Stranice
	for i in [1,3,5,7]:
		if je_slobodno_mesto(tabla,i):
			return i

Povlačenje poteza

  • Ako igra čovek, preuzimamo potez sa tastature
  • Ako igra računar, odredimo potez prema pravilima

def potez(tabla, znak, racunar):
	print 'igra '+znak
	
	if racunar == False:
		pozicija = preuzmi_poziciju()

  • Zatim igra računar

Iscrtavanje svih znakova na tabli

  • Prolazimo kroz tablu (listu)
  • Ukoliko naiđemo na "x" dodamo sliku x.png na koordinate koje odgovaraju toj toj poziciji (indeksu u listi)
  • Ukoliko naiđemo na "o" dodamo sliku o.png na koordinate koje odgovaraju toj toj poziciji (indeksu u listi) 

def iscrtaj_znakove(tabla, povrsina, x_img, o_img):
	for i in range(len(tabla)):
		if tabla[i]=="x":
			povrsina.blit(x_img, koordinate_pozicije(i))
		if tabla[i]=="o":
			povrsina.blit(o_img, koordinate_pozicije(i))
